query1: 日本の水面積率


SELECT distinct ?o
WHERE {dbpedia:日本 prop-ja:水面積率 ?o.}

query2: ProgrammingLanguage登場時期

select distinct * where {
?s rdf:type <http://dbpedia.org/ontology/ProgrammingLanguage>;
<http://ja.dbpedia.org/property/登場時期> ?o.}order by ?o

query3：日本の都道府県県庁所在地


select distinct ?s ?o where {
  ?s rdf:type dbpedia-owl:Place;
     dbpedia-owl:wikiPageWikiLink category-ja:日本の都道府県;
     prop-ja:所在地 ?o.
}

?s dbpedia-ja:ユービーアイソフト ?o.
?s dbpedia-owl:publisher dbpedia-ja:ユービーアイソフト.
}

query6:ps4またXboxONEで遊べるゲーム



SELECT DISTINCT ?game ?compony WHERE {
{?game dbpedia-owl:computingPlatform dbpedia-ja:PlayStation_4.
  }
UNION
{?game dbpedia-owl:computingPlatform dbpedia-ja:Xbox_One.}
  ?game dbpedia-owl:publisher ?compony.}
GROUP BY ?game
ORDER BY ?game

  ?s rdf:type schema:Mountain.

query8:日本の都道府県のdbpediaに載ってる有名人の数
SELECT ?s count(?o) AS ?c
WHERE { ?s rdf:type dbpedia-owl:Place;
     dbpedia-owl:wikiPageWikiLink category-ja:日本の都道府県.
     ?o dbpedia-owl:birthPlace ?s.
   }order by ?c
